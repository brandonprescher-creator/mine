## Database Schema Extensions

```sql
-- Activity assignments
CREATE TABLE activities (
    id INTEGER PRIMARY KEY,
    tool_type TEXT,
    title TEXT,
    description TEXT,
    lesson_id INTEGER,
    config JSON
);

-- Student submissions
CREATE TABLE submissions (
    id INTEGER PRIMARY KEY,
    activity_id INTEGER,
    student_id INTEGER,
    work_data JSON,
    submitted_at TIMESTAMP,
    status TEXT
);

-- Teacher feedback
CREATE TABLE feedback (
    id INTEGER PRIMARY KEY,
    submission_id INTEGER,
    teacher_id INTEGER,
    comments TEXT,
    score DECIMAL,
    rubric_data JSON
);

-- Student portfolios
CREATE TABLE portfolio_items (
    id INTEGER PRIMARY KEY,
    student_id INTEGER,
    submission_id INTEGER,
    reflection TEXT,
    featured BOOLEAN
);
```
